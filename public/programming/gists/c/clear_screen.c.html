Clear the terminal screen using ANSI control sequences.

#include <stdio.h>

// Clear screen using ANSI control sequences
// Supported by most Linux / Unix terminals and windows 10+ cmd
void clear_screen(void) {
    printf("\033[2J\033[H");
    fflush(stdout);
}
