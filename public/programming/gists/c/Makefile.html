C / Makefile

Simple Makefile template for C projects.

# Compiler flags
#
CFLAGS      = -Wall -Werror -Wextra -Wpedantic -std=c99
DEBUG_FLAGS = -ggdb
LDFLAGS     =
CC?         = gcc

#
# Project files
#
OUT_NAME = hello
OBJ      = hello.o

#
# Commands
#
all: $(OUT_NAME)

debug: CFLAGS += $(DEBUG_FLAGS)
debug: $(OUT_NAME)

run: $(OUT_NAME)
  chmod +x $(OUT_NAME)
  ./$(OUT_NAME)

clean:
  rm -f $(OBJ)

distclean:
  rm -f $(OUT_NAME)

$(OUT_NAME): $(OBJ)
  $(CC) $(OBJ) $(LDFLAGS) $(CFLAGS) -o $(OUT_NAME)

%.o: %.c
  $(CC) $(CFLAGS) -c $< -o $@
